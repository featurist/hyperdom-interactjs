plastiq-interactjs

A plastiq component that enables multi-touch manipulation of elements using interact.js. 

var plastiq = require('plastiq');
var interact = require('plastiq-interactjs');
var h = plastiq.html;

function render(model) {
  return h('.page',
    interact({ draggable: true },
      h('.red', 'Animal')
    ),
    interact({
        draggable: {
          snap: { targets: [ interact.createSnapGrid({ x: 20, y: 20 }) ] }
        }
      },
      h('.green', 'Vegetable')
    ),
    interact({ draggable: { inertia: true }, rotatable: true, scalable: true },
      h('.blue', 'Mineral')
    )
  );
}

plastiq.append(document.body, render, {});
